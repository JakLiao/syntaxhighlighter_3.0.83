删除应用名 --OPCFBLOG 要删除的表空间名
delete from maxapps where app='OPCFBLOG';
delete from maxpresentation where app='OPCFBLOG';
delete from sigoption where app='OPCFBLOG';
delete from applicationauth where app='OPCFBLOG';
delete from maxlabels where app='OPCFBLOG';
delete from maxmenu where moduleapp='OPCFBLOG' and menutype !='MODULE';
delete from maxmenu where moduleapp='OPCFBLOG' and elementtype='APP' and keyvalue='OPCFBLOG';
delete from maxmenu where keyvalue = 'OPCFBLOG';
delete from appdoctype where app= 'OPCFBLOG';

显示statusdate时间为2011年以后的
SELECT personid, status, displayname, department, title, statusdate
  FROM person
 where statusdate > =TO_DATE('2011-01-01', 'YYYY-MM-DD')
显示示statusdate时间为2011年6月的
SELECT personid, status, displayname, department, title, statusdate
  FROM person
 where statusdate >= TO_DATE('2011-06-01', 'YYYY-MM-DD') AND statusdate < TO_DATE('2011-7-1', 'YYYY-MM-DD')
